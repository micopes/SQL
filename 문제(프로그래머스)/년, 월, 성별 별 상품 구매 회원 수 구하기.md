``` SQL
WITH cte AS (
    SELECT 
        USER_ID,
        GENDER
    FROM USER_INFO
    WHERE GENDER IS NOT NULL
)

SELECT 
    YEAR(OS.SALES_DATE) AS YEAR,
    MONTH(OS.SALES_DATE) AS MONTH,
    UI.GENDER,
    COUNT(OS.USER_ID) AS USERS
FROM ONLINE_SALE OS
JOIN cte UI ON OS.USER_ID = UI.USER_ID
GROUP BY 1, 2, 3
ORDER BY 1, 2, 3
```
