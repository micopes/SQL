``` SQL
WITH cte AS (
    SELECT CAR_ID
    FROM CAR_RENTAL_COMPANY_RENTAL_HISTORY
    WHERE MONTH(START_DATE) IN (8, 9, 10)
    GROUP BY CAR_ID
    HAVING COUNT(*) >= 5
)

SELECT MONTH(START_DATE), CAR_ID, COUNT(*) AS RECORDS
FROM CAR_RENTAL_COMPANY_RENTAL_HISTORY
WHERE CAR_ID IN (SELECT CAR_ID FROM cte) AND MONTH(START_DATE) IN (8, 9, 10)
GROUP BY CAR_ID, 1
HAVING COUNT(*) >= 1
ORDER BY 1, 2 DESC
```
