``` SQL
WITH CTE AS (
    SELECT 
        CATEGORY,
        MAX(PRICE) AS MAX_PRICE
    FROM FOOD_PRODUCT
    GROUP BY CATEGORY
)

SELECT 
    F.CATEGORY, 
    F.PRICE AS MAX_PRICE, 
    F.PRODUCT_NAME
FROM FOOD_PRODUCT F, CTE F2
WHERE F.CATEGORY IN ('과자', '국', '김치', '식용유') 
    AND F.PRICE = F2.MAX_PRICE
    AND F.CATEGORY = F2.CATEGORY
ORDER BY 2 DESC
```
