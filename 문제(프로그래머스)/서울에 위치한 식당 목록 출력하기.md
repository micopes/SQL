``` SQL
SELECT 
    RI.REST_ID,
    RI.REST_NAME,
    RI.FOOD_TYPE,
    RI.FAVORITES,
    RI.ADDRESS,
    ROUND(AVG(RR.REVIEW_SCORE), 2) AS SCORE
FROM REST_INFO RI
JOIN REST_REVIEW RR ON RI.REST_ID = RR.REST_ID
WHERE LEFT(ADDRESS, 2) = '서울'
GROUP BY REST_ID
ORDER BY AVG(RR.REVIEW_SCORE) DESC, 4 DESC
```
